def quicksort(xs):
    if xs == []:
        return []
    else:
        smaller = [x for x in xs[1:] if x <= xs[0]]
        larger = [x for x in xs[1:] if x > xs[0]]
        return quicksort(smaller ) + [xs[0]] + quicksort(larger)
